const loadQuranData = async (recitation, apiUrl, storageKey) => {
  try {
    const response = await fetch(apiUrl);
    let data = await response.json();
    
    // التأكد من صحة البيانات
    if (data && !Array.isArray(data) && data.data) {
      data = data.data;
    }
    
    if (Array.isArray(data) && data.length === 114) {
      // تأكد من وجود الآيات
      const hasVerses = data[0]?.verses && data[0].verses.length > 0;
      if (hasVerses) {
        localStorage.setItem(storageKey, JSON.stringify(data));
        console.log(`✅ تم تحميل ${recitation} بنجاح - 114 سورة`);
        return true;
      }
    }
    throw new Error('بيانات غير صالحة');
  } catch (error) {
    console.error(`❌ فشل تحميل ${recitation}:`, error);
    return false;
  }
};

    if (Array.isArray(parsed) && parsed.length === 114) {

        name: ARABIC_SURAH_NAMES[idx] || surah.name

      setSurahs(parsed);
    }
  }

وفي HadithScreen.jsx، أصلح دالة loadBook لمعالجة صيغ JSON المختلفة:
javascript
const loadBook = async (bookId, bookName) => {
  setLoading(true);
  setError(null);
  try {
    let data = localStorage.getItem(`hadith_${bookId}`);
    if (!data) {
      const response = await fetch(`https://raw.githubusercontent.com/saandsoft/hadith-json/main/db/${bookId}.json`);
      data = await response.json();
      localStorage.setItem(`hadith_${bookId}`, JSON.stringify(data));
    } else {
      data = JSON.parse(data);
    }
    
    // استخراج الأحاديث من الصيغ المختلفة
    let hadithsList = [];
    if (data.hadiths && Array.isArray(data.hadiths)) {
      hadithsList = data.hadiths;
    } else if (Array.isArray(data)) {
      hadithsList = data;
    } else if (data.data && Array.isArray(data.data)) {
      hadithsList = data.data;
    }
    
    if (hadithsList.length > 0) {
      setHadiths(hadithsList);
      setSelectedBook(bookName);
    } else {
      setError(`لا توجد أحاديث في ${bookName} حالياً`);
      setHadiths([]);
    }
  } catch (error) {
    setError(`فشل تحميل ${bookName}. تأكد من الاتصال بالإنترنت.`);
    setHadiths([]);
  }
  setLoading(false);
};
