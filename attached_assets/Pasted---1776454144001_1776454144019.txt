javascript
import React, { useEffect, useState } from 'react';

const LoadingScreen = ({ onComplete }) => {
  const [progress, setProgress] = useState(0);
  const [status, setStatus] = useState('');

  useEffect(() => {
    // التحقق من صحة البيانات الموجودة
    const isValidData = () => {
      const hafs = localStorage.getItem('quran_hafs');
      if (!hafs) return false;
      try {
        const parsed = JSON.parse(hafs);
        return parsed && parsed.length === 114 && parsed[0]?.verses?.length > 0;
      } catch { return false; }
    };

    if (isValidData()) {
      onComplete();
      return;
    }
    loadAllData();
  }, []);

  const loadQuranData = async (recitation, url) => {
    try {
      const res = await fetch(url);
      let data = await res.json();
      if (data && !Array.isArray(data) && data.data) data = data.data;
      if (data && Array.isArray(data) && data.length === 114) {
        localStorage.setItem(`quran_${recitation}`, JSON.stringify(data));
        return true;
      }
      return false;
    } catch (e) { return false; }
  };

  const loadHadithBook = async (id, name) => {
    try {
      const res = await fetch(`https://raw.githubusercontent.com/saandsoft/hadith-json/main/db/${id}.json`);
      const data = await res.json();
      localStorage.setItem(`hadith_${id}`, JSON.stringify(data));
      return true;
    } catch (e) { return false; }
  };

  const loadAllData = async () => {
    setStatus('جاري تحميل القرآن - رواية حفص...');
    await loadQuranData('hafs', 'https://raw.githubusercontent.com/risan/quran-json/main/data/quran.json');
    setProgress(15);

    setStatus('جاري تحميل القرآن - رواية ورش...');
    await loadQuranData('warsh', 'https://raw.githubusercontent.com/risan/quran-json/main/data/quran-warsh.json');
    setProgress(30);

    const books = ['bukhari','muslim','tirmidhi','abudawud','nasai','ibnmajah','malik','ahmad','nawawi','riyadh'];
    for (let i = 0; i < books.length; i++) {
      setStatus(`جاري تحميل كتاب الحديث ${i+1}/${books.length}...`);
      await loadHadithBook(books[i], '');
      setProgress(30 + (i+1) * 5);
    }

    setProgress(100);
    setStatus('تم التحميل بنجاح!');
    setTimeout(onComplete, 1500);
  };

  return (
    <div className="flex flex-col items-center justify-center min-h-screen bg-background">
      <h1 className="text-5xl font-bold text-gold mb-4">رفيق الدرب</h1>
      <div className="w-64 h-2 bg-gray-700 rounded-full overflow-hidden">
        <div className="h-full bg-gold transition-all duration-300" style={{ width: `${progress}%` }} />
      </div>
      <p className="text-white mt-4">{status}</p>
      <p className="text-gray-500 text-sm mt-4">يحتاج إنترنت لأول مرة فقط</p>
    </div>
  );
};

export default LoadingScreen;

const startListening = () => {
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  const recognition = new SpeechRecognition();
  recognition.lang = 'ar-SA';
  recognition.onresult = (event) => {
    const spoken = event.results[0][0].transcript;
    setRecognizedText(spoken);
    const expected = selectedSurah.verses[currentAyah]?.text;
    if (spoken !== expected) {
      if (navigator.vibrate) navigator.vibrate(500);
      let errorMsg = 'خطأ في التلاوة';
      if (expected.includes('مَنْ آمَنَ') && !spoken.includes('مَنْ آمَنَ'))
        errorMsg = '⚠️ خطأ في الإظهار: يجب إظهار النون الساكنة قبل الهمزة';
      else if (expected.includes('مِن رَّبِّ') && !spoken.includes('مِن رَّبِّ'))
        errorMsg = '⚠️ خطأ في الإدغام: يجب إدغام النون الساكنة في الراء';
      setCorrection({ spoken, expected, errorMsg });
      new SpeechSynthesisUtterance(expected).speak();
    } else {
      alert('✅ أحسنت!');
      nextAyah();
    }
  };
  recognition.start();
};

بوصلة القبلة:
javascript
const getQibla = (lat, lng) => {
  // معادلة حساب زاوية القبلة من مكة
  const meccaLat = 21.4225, meccaLng = 39.8262;
  const φ1 = lat * Math.PI/180, φ2 = meccaLat * Math.PI/180;
  const Δλ = (meccaLng - lng) * Math.PI/180;
  const y = Math.sin(Δλ) * Math.cos(φ2);
  const x = Math.cos(φ1)*Math.sin(φ2) - Math.sin(φ1)*Math.cos(φ2)*Math.cos(Δλ);
  let θ = Math.atan2(y, x);
  return (θ * 180/Math.PI + 360) % 360;
};
